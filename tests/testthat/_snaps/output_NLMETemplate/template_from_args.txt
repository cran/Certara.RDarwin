##Description: A basic 1-cpt PK model test
##Author: R Test Suite
##DATA {data_dir}/test_data.csv
##MAP    A1 = AMT CObs = DV id = ID time = TIME
##MODEL test() {
	cfMicro(A1, Cl / V)
	C = A1 / V
	dosepoint(A1, idosevar = A1Dose, infdosevar = A1InfDose, infratevar = A1InfRate)
	error(CEps = 0.1)
	observe(CObs = C * (1 + CEps))
	
	stparm(Cl = tvCl * exp( nCl ))
	fixef(tvCl= c(, 1, ))
	ranef(diag(nCl) = c(0.09))
	stparm(V = tvV * exp( nV ))
	fixef(tvV= c(, 10, ))
	ranef(diag(nV) = c(0.09))

}
